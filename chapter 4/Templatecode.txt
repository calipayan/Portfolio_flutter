// Simplified Recipes Page with Add Recipe Feature
class RecipesPage extends StatefulWidget {
  @override
  _RecipesPageState createState() => _RecipesPageState();
}

class _RecipesPageState extends State<RecipesPage> {
  final List<Map<String, String>> recipes = [
    {
      'name': 'Spaghetti Carbonara',
      'description': 'A creamy pasta with eggs, cheese, and bacon.',
    },
    {
      'name': 'Chicken Alfredo',
      'description': 'A rich and creamy pasta dish with chicken and parmesan.',
    },
    {
      'name': 'Vegetable Stir-fry',
      'description': 'A healthy mix of stir-fried veggies with soy sauce.',
    },
    {
      'name': 'Beef Tacos',
      'description': 'Delicious tacos filled with seasoned beef and toppings.',
    },
  ];

  final TextEditingController _nameController = TextEditingController();
  final TextEditingController _descriptionController = TextEditingController();

  // Function to add a new recipe
  void _addRecipe() {
    if (_nameController.text.isNotEmpty && _descriptionController.text.isNotEmpty) {
      setState(() {
        recipes.add({
          'name': _nameController.text,
          'description': _descriptionController.text,
        });
      });
      _nameController.clear();
      _descriptionController.clear();
      Navigator.pop(context); // Close the dialog after adding the recipe
    }
  }

  // Dialog to add a new recipe
  void _showAddRecipeDialog() {
    showDialog(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: Text('Add New Recipe'),
          content: Column(
            mainAxisSize: MainAxisSize.min,
            children: [
              TextField(
                controller: _nameController,
                decoration: InputDecoration(hintText: 'Enter Recipe Name'),
              ),
              SizedBox(height: 10),
              TextField(
                controller: _descriptionController,
                decoration: InputDecoration(hintText: 'Enter Recipe Description'),
                maxLines: 4,
              ),
            ],
          ),
          actions: [
            TextButton(
              onPressed: () {
                _addRecipe();
              },
              child: Text('Add Recipe'),
            ),
            TextButton(
              onPressed: () {
                Navigator.pop(context); // Close dialog if cancel
              },
              child: Text('Cancel'),
            ),
          ],
        );
      },
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Recipes'),
        backgroundColor: Colors.brown.shade300, // Light brown background for visibility
      ),
      body: ListView.builder(
        itemCount: recipes.length,
        itemBuilder: (context, index) {
          return ListTile(
            title: Text(
              recipes[index]['name']!,
              style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold),
            ),
            subtitle: Text(
              recipes[index]['description']!,
              style: TextStyle(fontSize: 16),
            ),
            onTap: () {
              // Show simple dialog for recipe detail
              showDialog(
                context: context,
                builder: (context) => AlertDialog(
                  title: Text(recipes[index]['name']!),
                  content: Text('Instructions: Cook the recipe!'),
                  actions: [
                    TextButton(
                      onPressed: () {
                        Navigator.pop(context);
                      },
                      child: Text('Close'),
                    ),
                  ],
                ),
              );
            },
          );
        },
      ),
      // Floating action button to add a recipe
      floatingActionButton: FloatingActionButton(
        onPressed: _showAddRecipeDialog,
        tooltip: 'Add Recipe',
        child: Icon(Icons.add),
      ),
    );
  }
}
